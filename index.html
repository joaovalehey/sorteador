  <script>


    let numerosSorteados = [];

    let primeiroSorteio = true;


    function sortearNumero(){


      const min = parseInt(document.getElementById("min").value);

      const max = parseInt(document.getElementById("max").value);

      const naoRepetir = document.getElementById("naoRepetir").checked;


      if(min >= max){

        alert("O número inicial deve ser menor que o final.");

        return;

      }


      let numero;


      // PRIMEIRO SORTEIO SEMPRE SERÁ 2

      if(primeiroSorteio){


        numero = 2;

        primeiroSorteio = false;


      }else{


        if(naoRepetir){


          const totalPossivel = max - min + 1;


          if(numerosSorteados.length >= totalPossivel){

            alert("Todos os números já foram sorteados!");

            return;

          }


          do{

            numero = gerarNumero(min,max);

          }while(numerosSorteados.includes(numero));


        }else{

          numero = gerarNumero(min,max);

        }

      }


      numerosSorteados.push(numero);


      document.getElementById("resultado").style.display = "block";

      document.getElementById("numero").innerText = numero;


      atualizarHistorico();

    }


    function gerarNumero(min,max){

      return Math.floor(Math.random() * (max - min + 1)) + min;

    }


    function atualizarHistorico(){


      const lista = document.getElementById("lista");


      lista.innerHTML = "";


      numerosSorteados

        .slice()

        .reverse()

        .forEach(num => {


          const item = document.createElement("div");

          item.classList.add("item");

          item.innerText = num;


          lista.appendChild(item);

        });

    }


  </script>
